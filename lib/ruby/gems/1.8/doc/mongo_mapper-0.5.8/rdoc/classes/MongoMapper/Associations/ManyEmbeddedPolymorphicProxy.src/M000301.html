      def replace(v)
        @_values = v.map do |doc_or_hash|
          if doc_or_hash.kind_of?(EmbeddedDocument)
            doc = doc_or_hash
            {@association.type_key_name => doc.class.name}.merge(doc.attributes)
          else
            doc_or_hash
          end
        end
        
        reset
      end
