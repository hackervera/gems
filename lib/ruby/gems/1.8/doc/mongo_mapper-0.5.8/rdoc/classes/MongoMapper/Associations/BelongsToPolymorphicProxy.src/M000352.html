      def replace(doc)
        if doc
          doc.save if doc.new?
          id, type = doc.id, doc.class.name
        end
        
        @owner.send("#{@association.foreign_key}=", id)
        @owner.send("#{@association.type_key_name}=", type)
        reset
      end
