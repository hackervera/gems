        def initialize(nodes_or_host=nil, port=nil)
          @nodes = case nodes_or_host
                   when String
                     [[nodes_or_host, port || DEFAULT_PORT]]
                   when Array
                     nodes_or_host.collect { |nh| [nh[0], nh[1] || DEFAULT_PORT] }
                   when nil
                     [['localhost', DEFAULT_PORT]]
                   end
        end
