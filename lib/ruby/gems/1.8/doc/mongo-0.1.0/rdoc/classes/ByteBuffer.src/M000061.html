  def put_long(i, offset=nil)
    offset = @cursor unless offset
    if @int_pack_order == 'N'
      put_int(i >> 32, offset)
      put_int(i & 0xffffffff, offset + 4)
    else
      put_int(i & 0xffffffff, offset)
      put_int(i >> 32, offset + 4)
    end
  end
