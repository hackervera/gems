BYTE_ORDER = [7, 6, 5, 4, 3, 2, 1, 0, 11, 10, 9, 8]   This is the legacy byte ordering for Babble. Versions of the Ruby driver prior to 0.14 used this byte ordering when converting ObjectID instances to and from strings. If you have string representations of ObjectIDs using the legacy byte ordering make sure to use the to_s_legacy and from_string_legacy methods, or convert your strings with ObjectID#legacy_string_convert

    def self.from_string(str)
      raise "illegal ObjectID format" unless legal?(str)
      data = []
      12.times do |i|
        data[i] = str[i * 2, 2].to_i(16)
      end
      self.new(data)
    end

      raise "illegal ObjectID format" unless legal?(str)

      BYTE_ORDER.each_with_index { |string_position, data_index|

    def self.legacy_string_convert(str)
      legacy = ' ' * 24
      BYTE_ORDER.each_with_index do |legacy_pos, pos|
        legacy[legacy_pos * 2, 2] = str[pos * 2, 2]
      end
      legacy
    end

    def self.legal?(str)
      len = BYTE_ORDER.length * 2
      str =~ /([0-9a-f]+)/i
      match = $1
      str && str.length == len && match == str
    end
