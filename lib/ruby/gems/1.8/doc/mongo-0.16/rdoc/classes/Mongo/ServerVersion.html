    def <=>(new)
      local, new  = self.to_a, to_array(new)
      for n in 0...local.size do 
        break if elements_include_mods?(local[n], new[n])
        if local[n] < new[n].to_i
          result = -1
          break;
        elsif local[n] > new[n].to_i
          result = 1
          break;
        end
      end
      result || 0
    end
